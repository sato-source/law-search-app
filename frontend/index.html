function formatLawXml(xml){ if(!xml)return "" const parser=new DOMParser() const doc=parser.parseFromString(xml,"application/xml") if(doc.querySelector("parsererror")){ return stripXml(xml) } const lines=[] const lawTitle= doc.querySelector("LawTitle")?.textContent?.trim()|| doc.querySelector("LawNum")?.textContent?.trim()|| "" if(lawTitle){ lines.push(lawTitle) lines.push("") } const main=doc.querySelector("MainProvision") if(!main){ return stripXml(xml) } collectNode(main,lines) return lines.join("\n") } function collectNode(node,lines){ for(const child of node.children){ const tag=child.tagName if(tag==="Part"||tag==="Chapter"||tag==="Section"){ const title= child.querySelector("PartTitle,ChapterTitle,SectionTitle") ?.textContent?.trim() if(title){ lines.push(title) lines.push("") } collectNode(child,lines) continue } if(tag==="Article"){ const num=child.getAttribute("Num")||"" lines.push(`第${num}条`) const caption= child.querySelector("ArticleCaption") ?.textContent?.trim() if(caption){ lines.push("【"+caption+"】") } const paras=child.querySelectorAll(":scope > Paragraph") paras.forEach(p=>collectParagraph(p,lines)) lines.push("") continue } collectNode(child,lines) } } function collectParagraph(p,lines){ const num=p.getAttribute("Num")||"" let text="" const sentence=p.querySelector("ParagraphSentence") if(sentence){ text=sentence.textContent.trim() }else{ text=p.textContent.trim() } if(text){ if(num && num!=="1"){ lines.push(num+" "+text) }else{ lines.push(text) } } const items=p.querySelectorAll(":scope > Item") items.forEach(item=>{ const itemTitle=item.querySelector("ItemTitle")?.textContent.trim()||"" const itemText=item.querySelector("ItemSentence") ?.textContent.trim() ||item.textContent.trim() if(itemTitle){ lines.push(itemTitle+" "+itemText) }else{ lines.push(itemText) } }) } function stripXml(xml){ return xml .replace(/<[^>]+>/g,"\n") .replace(/\n+/g,"\n") .trim() }
